# Better Admin Components

Better Admin provides **78 production-ready components** organized into **10 categories**. All components are designed to work seamlessly with **better-query** and **better-auth**, following the patterns from [shadcn-admin-kit](https://marmelab.com/shadcn-admin-kit/) while being completely independent of react-admin (ra-core).

## Overview

Unlike react-admin, Better Admin uses:
- **better-query** for all data operations (not ra-data-simple-rest)
- **better-auth** for authentication (not ra-core auth)
- **shadcn/ui** components for UI (not Material-UI)
- **Direct database access** (no REST API required)

## Installation

Install any component using the Better Admin CLI:

```bash title="Terminal"
npx better-admin add [component-name]
```

The CLI automatically:
- Installs required shadcn/ui components
- Installs npm dependencies
- Copies the component to your project

### Examples

```bash title="Terminal"
# Install data table
npx better-admin add data-table

# Install CRUD form
npx better-admin add crud-form

# Install authentication
npx better-admin add login-page
```

          await update.mutateAsync({ where: { id }, data: formData });
        } else {
          await create.mutateAsync(formData);
        }
      }}
    />
  );
}

## Complete Usage Examples

### Building a User Management Interface

Here's a complete example using better-query for CRUD operations:

```tsx title="app/admin/users/page.tsx"
"use client";

import { useBetterQuery } from "better-admin";
import { query } from "@/lib/query";
import { DataTable } from "@/components/admin/data-table";
import { CreateButton } from "@/components/admin/create-button";
import { EditButton } from "@/components/admin/edit-button";
import { DeleteButton } from "@/components/admin/delete-button";

export default function UsersListPage() {
  const { list, remove } = useBetterQuery("user", query);
  const { data, isLoading, refetch } = list.useQuery();

  if (isLoading) return <div>Loading...</div>;

  const columns = [
    { accessorKey: "name", header: "Name" },
    { accessorKey: "email", header: "Email" },
    { accessorKey: "role", header: "Role" },
    {
      id: "actions",
      header: "Actions",
      cell: ({ row }) => (
        <div className="flex gap-2">
          <EditButton id={row.original.id} />
          <DeleteButton
            onConfirm={async () => {
              await remove.mutateAsync({ where: { id: row.original.id } });
              refetch();
            }}
          />
        </div>
      ),
    },
  ];

  return (
    <div className="space-y-4">
      <div className="flex justify-between">
        <h1 className="text-2xl font-bold">Users</h1>
        <CreateButton href="/admin/users/create" />
      </div>
      <DataTable columns={columns} data={data || []} />
    </div>
  );
}
```

### List with Filters and Search

```tsx title="app/admin/users/page.tsx"
"use client";

import { useState } from "react";
import { useBetterQuery } from "better-admin";
import { query } from "@/lib/query";
import { DataTable } from "@/components/admin/data-table";
import { Input } from "@/components/ui/input";
import { Select } from "@/components/ui/select";

export default function UsersListWithFilters() {
  const [search, setSearch] = useState("");
  const [role, setRole] = useState<string>("");
  const { list } = useBetterQuery("user", query);

  // Build where clause based on filters
  const where: any = {};
  if (search) {
    where.OR = [
      { name: { contains: search } },
      { email: { contains: search } },
    ];
  }
  if (role) {
    where.role = role;
  }

  const { data, isLoading } = list.useQuery({ where });

  const columns = [
    { accessorKey: "name", header: "Name" },
    { accessorKey: "email", header: "Email" },
    { accessorKey: "role", header: "Role" },
  ];

  return (
    <div className="space-y-4">
      <div className="flex gap-4">
        <Input
          placeholder="Search by name or email..."
          value={search}
          onChange={(e) => setSearch(e.target.value)}
          className="max-w-sm"
        />
        <Select
          value={role}
          onValueChange={setRole}
        >
          <option value="">All Roles</option>
          <option value="user">User</option>
          <option value="admin">Admin</option>
        </Select>
      </div>
      {isLoading ? (
        <div>Loading...</div>
      ) : (
        <DataTable columns={columns} data={data || []} />
      )}
    </div>
  );
}
```

## Authentication Integration

All components work seamlessly with better-auth:

```tsx title="app/admin/layout.tsx"
"use client";

import { useEffect } from "react";
import { useRouter } from "next/navigation";
import { useBetterAuth } from "better-admin";
import { authClient } from "@/lib/auth-client";
import { UserMenu } from "@/components/admin/user-menu";

export default function AdminLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  const { user, isLoading, signOut } = useBetterAuth(authClient);
  const router = useRouter();

  useEffect(() => {
    if (!isLoading && !user) {
      router.push("/login");
    }
  }, [user, isLoading, router]);

  if (isLoading) return <div>Loading...</div>;
  if (!user) return null;

  return (
    <div className="min-h-screen bg-background">
      <header className="border-b">
        <div className="flex h-16 items-center px-4 justify-between">
          <h1 className="text-xl font-bold">Admin Portal</h1>
          <UserMenu user={user} onSignOut={signOut} />
        </div>
      </header>
      <main className="p-6">{children}</main>
    </div>
  );
}
```

## Best Practices

### 1. Type Safety with better-query

Always define your resource schemas for full type safety:

```typescript title="lib/query.ts"
import { betterQuery, createResource } from "better-query";

const userResource = createResource({
  name: "user",
  schema: {
    id: { type: "string", primary: true },
    name: { type: "string", required: true },
    email: { type: "string", required: true },
    role: { type: "string" },
    createdAt: { type: "date" },
    updatedAt: { type: "date" },
  },
});

export const query = betterQuery({
  database: db,
  resources: [userResource],
});
```

### 2. Error Handling

Always handle errors gracefully:

```tsx
const { create } = useBetterQuery("user", query);

const handleSubmit = async (data: any) => {
  try {
    await create.mutateAsync(data);
    toast.success("User created successfully");
    router.push("/admin/users");
  } catch (error) {
    toast.error("Failed to create user");
    console.error(error);
  }
};
```

### 3. Loading States

Show loading indicators for better UX:

```tsx
const { data, isLoading, error } = list.useQuery();

if (isLoading) return <Spinner />;
if (error) return <Error message={error.message} />;
if (!data) return <div>No data found</div>;

return <DataTable data={data} />;
```

### 4. Optimistic Updates

Use optimistic updates for instant feedback:

```tsx
const { update } = useBetterQuery("user", query);

const toggleActive = async (id: string, active: boolean) => {
  // Optimistic update
  queryClient.setQueryData(["users", id], (old: any) => ({
    ...old,
    active: !active,
  }));

  try {
    await update.mutateAsync({
      where: { id },
      data: { active: !active },
    });
  } catch (error) {
    // Rollback on error
    queryClient.invalidateQueries(["users", id]);
  }
};
```

### 5. Permissions

Check permissions before showing actions:

```tsx
import { useBetterAuth } from "better-admin";
import { authClient } from "@/lib/auth-client";

export function UserActions({ user }: { user: any }) {
  const { user: currentUser } = useBetterAuth(authClient);
  const canEdit = currentUser?.role === "admin" || currentUser?.id === user.id;
  const canDelete = currentUser?.role === "admin";

  return (
    <div className="flex gap-2">
      {canEdit && <EditButton id={user.id} />}
      {canDelete && <DeleteButton onConfirm={() => deleteUser(user.id)} />}
    </div>
  );
}
```

### 6. Debounce Search

Prevent excessive queries with debouncing:

```tsx
import { useState, useEffect } from "react";
import { useDebouncedValue } from "@/hooks/use-debounced-value";

export function SearchableList() {
  const [search, setSearch] = useState("");
  const debouncedSearch = useDebouncedValue(search, 300);
  const { list } = useBetterQuery("user", query);

  const { data } = list.useQuery({
    where: debouncedSearch ? { name: { contains: debouncedSearch } } : {},
  });

  return (
    <div>
      <Input
        placeholder="Search..."
        value={search}
        onChange={(e) => setSearch(e.target.value)}
      />
      <DataTable data={data || []} />
    </div>
  );
}
```

      options: [
        { value: "user", label: "User" },
        { value: "admin", label: "Admin" },
      ],

    <CrudForm
      fields={fields}
      defaultValues={data}
      onSubmit={async (formData) => {

          await update.mutateAsync({ where: { id }, data: formData });

    />
