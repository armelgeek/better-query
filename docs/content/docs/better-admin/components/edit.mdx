# Edit

The `<Edit>` component is the page component for the edit view. It renders a default layout (title, actions) and delegates the actual form rendering to its children (usually a `<SimpleForm>`).

## Usage

```tsx
import { Edit, SimpleForm, TextInput } from '@/components/admin';

export const PostEdit = () => (
  <Edit>
    <SimpleForm>
      <TextInput source="title" required />
      <TextInput source="content" multiline />
      <SelectInput
        source="status"
        choices={[
          { id: 'draft', name: 'Draft' },
          { id: 'published', name: 'Published' },
        ]}
      />
    </SimpleForm>
  </Edit>
);
```

## Installation

```bash title="Terminal"
npx better-admin add edit
```

## Props

| Prop | Required | Type | Default | Description |
|------|----------|------|---------|-------------|
| `children` | Required | `ReactNode` | - | Form component (e.g. `<SimpleForm>`) |
| `title` | Optional | `string \| ReactNode` | 'Edit [Resource]' | Page title |
| `actions` | Optional | `ReactNode` | default actions | Custom action buttons |
| `redirect` | Optional | `string \| false` | 'list' | Where to redirect after save |
| `resource` | Optional | `string` | inferred | Resource name |
| `id` | Optional | `string \| number` | from URL | Record ID to edit |
| `transform` | Optional | `(data) => data` | - | Transform data before submission |
| `mutationMode` | Optional | `'pessimistic' \| 'optimistic' \| 'undoable'` | 'pessimistic' | Mutation mode |

## With better-query

Edit automatically fetches and updates data using better-query:

```tsx
import { Edit, SimpleForm, TextInput } from '@/components/admin';

export const UserEdit = () => (
  <Edit>

      <SelectInput
        source="role"
        choices={[
          { id: 'user', name: 'User' },
          { id: 'admin', name: 'Admin' },
        ]}
      />
    </SimpleForm>
  </Edit>
);
```

## Custom Title

Display the record in the title:

```tsx
import { Edit, SimpleForm, TextInput, useRecordContext } from '@/components/admin';

const PostTitle = () => {
  const record = useRecordContext();
  return <span>Edit "{record?.title}"</span>;
};

export const PostEdit = () => (
  <Edit title={<PostTitle />}>
    <SimpleForm>
      <TextInput source="title" />
      <TextInput source="content" />
    </SimpleForm>
  </Edit>
);
```

## Custom Actions

Override the default action buttons:

```tsx
import { Edit, SaveButton, DeleteButton, ShowButton } from '@/components/admin';

<Edit
  actions={
    <div className="flex gap-2">
      <ShowButton />
      <DeleteButton />
      <SaveButton />
    </div>
  }
>
  <SimpleForm>
    {/* inputs */}
  </SimpleForm>
</Edit>
```

## Redirect After Save

Control where users go after saving:

```tsx
// Redirect to the show page
<Edit redirect="show">
  <SimpleForm>{/* inputs */}</SimpleForm>
</Edit>

// Stay on the edit page
<Edit redirect={false}>
  <SimpleForm>{/* inputs */}</SimpleForm>
</Edit>

// Custom redirect
<Edit redirect={(resource, id) => `/custom/${resource}/${id}`}>
  <SimpleForm>{/* inputs */}</SimpleForm>
</Edit>
```

## Transform Data Before Save

Modify the data before sending it to the server:

```tsx
<Edit
  transform={(data) => ({
    ...data,
    updated_at: new Date().toISOString(),
    updated_by: getCurrentUserId(),
  })}
>
  <SimpleForm>
    <TextInput source="title" />
    <TextInput source="content" />
  </SimpleForm>
</Edit>
```

## Mutation Modes

Control how updates are handled:

```tsx
// Pessimistic: Wait for server response before updating UI
<Edit mutationMode="pessimistic">
  <SimpleForm>{/* inputs */}</SimpleForm>
</Edit>

// Optimistic: Update UI immediately, rollback on error
<Edit mutationMode="optimistic">
  <SimpleForm>{/* inputs */}</SimpleForm>
</Edit>

// Undoable: Update UI immediately, allow undo within timeout
<Edit mutationMode="undoable">
  <SimpleForm>{/* inputs */}</SimpleForm>
</Edit>
```

## Validation

Add validation to inputs:

```tsx
import { Edit, SimpleForm, TextInput, required, email, minLength } from '@/components/admin';

export const UserEdit = () => (
  <Edit>
    <SimpleForm>
      <TextInput source="name" validate={[required(), minLength(3)]} />
      <TextInput source="email" validate={[required(), email()]} />
      <TextInput
        source="age"
        validate={(value) => {
          if (value < 18) return 'Must be 18 or older';
        }}
      />
    </SimpleForm>
  </Edit>
);
```

## Disable Inputs

Conditionally disable inputs:

```tsx
import { Edit, SimpleForm, TextInput, useRecordContext } from '@/components/admin';

export const PostEdit = () => {
  const record = useRecordContext();
  
  return (
    <Edit>
      <SimpleForm>
        <TextInput source="title" />
        <TextInput 
          source="slug" 
          disabled={record?.published}
          helperText="Cannot edit slug after publishing"
        />
        <SelectInput source="status" />
      </SimpleForm>
    </Edit>
  );
};
```

## Dependent Inputs

Make inputs depend on other input values:

```tsx
import { Edit, SimpleForm, TextInput, SelectInput, FormDataConsumer } from '@/components/admin';

export const ProductEdit = () => (
  <Edit>
    <SimpleForm>
      <SelectInput
        source="type"
        choices={[
          { id: 'physical', name: 'Physical' },
          { id: 'digital', name: 'Digital' },
        ]}
      />
      <FormDataConsumer>
        {({ formData }) =>
          formData.type === 'physical' && (
            <>
              <TextInput source="weight" />
              <TextInput source="dimensions" />
            </>
          )
        }
      </FormDataConsumer>
    </SimpleForm>
  </Edit>
);
```

## Complete Example

```tsx
import {
  Edit,
  SimpleForm,
  TextInput,
  ReferenceInput,
  AutocompleteInput,
  SelectInput,
  BooleanInput,
  DateInput,
  FileInput,
  ImageField,
  DeleteButton,
  SaveButton,
  required,
  minLength,
  useRecordContext,
} from '@/components/admin';

const PostTitle = () => {
  const record = useRecordContext();
  return <span>Edit "{record?.title}"</span>;
};

export const PostEdit = () => (
  <Edit
    title={<PostTitle />}
    redirect="show"
    mutationMode="optimistic"
    transform={(data) => ({
      ...data,
      updated_at: new Date().toISOString(),
    })}
    actions={
      <div className="flex gap-2">
        <DeleteButton />
        <SaveButton />
      </div>
    }
  >
    <SimpleForm>
      <TextInput 
        source="title" 
        validate={[required(), minLength(5)]}
        fullWidth
      />
      <TextInput 
        source="content" 
        multiline
        rows={10}
        fullWidth
      />
      <ReferenceInput source="author_id" reference="users">
        <AutocompleteInput optionText="name" />
      </ReferenceInput>
      <SelectInput
        source="status"
        choices={[
          { id: 'draft', name: 'Draft' },
          { id: 'published', name: 'Published' },
        ]}
      />
      <BooleanInput source="featured" />
      <DateInput source="publish_date" />
      <FileInput source="cover_image" accept="image/*">
        <ImageField source="src" title="title" />
      </FileInput>
    </SimpleForm>
  </Edit>
);
```

## Integration with better-auth

Track who updated records:

```tsx
import { Edit, SimpleForm, TextInput } from '@/components/admin';
import { useBetterAuth } from 'better-admin';
import { authClient } from '@/lib/auth-client';

export const PostEdit = () => {
  const { user } = useBetterAuth(authClient);

  return (
    <Edit
      transform={(data) => ({
        ...data,
        updated_by: user?.id,
      })}
    >
      <SimpleForm>
        <TextInput source="title" />
        <TextInput source="content" />
      </SimpleForm>
    </Edit>
  );
};
```

## Loading State

The Edit component handles loading automatically:

```tsx
export const PostEdit = () => (
  <Edit>
    <SimpleForm>
      {/* Form inputs are disabled while loading */}
      <TextInput source="title" />
    </SimpleForm>
  </Edit>
);
```

## Error Handling

Errors are displayed automatically:

```tsx
export const PostEdit = () => (
  <Edit>
    <SimpleForm>
      {/* Validation errors are shown on inputs */}
      {/* Server errors are shown in a notification */}
      <TextInput source="title" />
    </SimpleForm>
  </Edit>
);
```
