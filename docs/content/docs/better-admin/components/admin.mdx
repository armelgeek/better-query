# Admin

The `Admin` component is the root wrapper for your Better Admin application. It provides context for authentication, data providers, and global configuration.

## Usage

```tsx
import { Admin } from '@/components/admin';
import { createBetterAuthProvider } from 'better-admin';
import { authClient } from '@/lib/auth-client';

const authProvider = createBetterAuthProvider({ authClient });

export default function App() {
  return (
    <Admin authProvider={authProvider}>
      <Resource name="posts" list={PostList} edit={PostEdit} create={PostCreate} />
      <Resource name="users" list={UserList} edit={UserEdit} />
    </Admin>
  );
}
```

## Props

| Prop | Type | Description |
|------|------|-------------|
| `authProvider` | `AuthProvider` | Authentication provider instance |
| `dataProvider` | `DataProvider` | Data provider for API calls |
| `theme` | `'light' \| 'dark' \| 'system'` | Theme configuration |
| `children` | `ReactNode` | Resource components and other content |

## Examples

### Basic Setup

```tsx
import { Admin, Resource } from '@/components/admin';

export default function App() {
  return (
    <Admin>
      <Resource name="posts" list={PostList} />
    </Admin>
  );
}
```

### With Authentication

```tsx
import { Admin } from '@/components/admin';
import { createBetterAuthProvider } from 'better-admin';
import { authClient } from '@/lib/auth-client';

const authProvider = createBetterAuthProvider({ authClient });

export default function App() {
  return (
    <Admin authProvider={authProvider}>
      <Resource name="posts" list={PostList} edit={PostEdit} />
    </Admin>
  );
}
```

### With Custom Theme

```tsx
import { Admin } from '@/components/admin';

export default function App() {
  return (
    <Admin theme="dark">
      <Resource name="posts" list={PostList} />
    </Admin>
  );
}
```
